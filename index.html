async function aggiornaSito() { try { // Cache buster per forzare l'aggiornamento dei dati const r = await fetch('risultati_v4.json?t=' + Date.now()); const risultati = await r.json(); const grid = document.getElementById('container-ruote'); grid.innerHTML = ''; Object.keys(risultati).forEach(ruota => { const dati = risultati[ruota]; const card = `
${ruota}
SCORE: ${dati.score}
${dati.ambo[0]} ${dati.ambo[1]}
DIAMETRALI D45:
${dati.diametrali.join(' - ')}
Rimanenti: ${dati.countdown} estr.
`; grid.innerHTML += card; }); } catch (err) { console.error("Errore caricamento:", err); } } // Avvia al caricamento della pagina window.onload = aggiornaSito;
